df = pd.read_csv('netflix_movies__1_.csv')

# Parse dates & extract helper columns
df['date_added']  = pd.to_datetime(df['date_added'].str.strip(), errors='coerce')
df['year_added']  = df['date_added'].dt.year
df['month_added'] = df['date_added'].dt.month

# Split by type
movies = df[df['type'] == 'Movie'].copy()
shows  = df[df['type'] == 'TV Show'].copy()

# Parse numeric duration
movies['mins']   = movies['duration'].str.extract(r'(\d+)').astype(float)
shows['seasons'] = shows['duration'].str.extract(r'(\d+)').astype(float)

print(f"Total titles : {len(df):,}")
print(f"  Movies     : {len(movies):,}  ({len(movies)/len(df)*100:.1f}%)")
print(f"  TV Shows   : {len(shows):,}  ({len(shows)/len(df)*100:.1f}%)")
print(f"\nRelease year range : {df['release_year'].min()} – {df['release_year'].max()}")
print(f"Missing values:\n{df.isnull().sum()[df.isnull().sum() > 0].to_string()}")

def get_genres(sub_df):
    """Flatten comma-separated 'listed_in' column into a genre Counter."""
    genres = []
    for row in sub_df['listed_in'].dropna():
        genres.extend([x.strip() for x in row.split(',')])
    return Counter(genres)

movie_genres = get_genres(movies)
show_genres  = get_genres(shows)
top_movie_g  = dict(movie_genres.most_common(10))
top_show_g   = dict(show_genres.most_common(10))

print("Top 10 Movie Genres:")
for g, c in top_movie_g.items():
    print(f"  {g:<30} {c:>4}")

print("\nTop 10 TV Show Genres:")
for g, c in top_show_g.items():
    print(f"  {g:<30} {c:>4}")

# Titles added to Netflix per year (from date_added)
trend = (
    df.groupby(['year_added', 'type'])
    .size()
    .unstack(fill_value=0)
)
trend = trend[trend.index >= 2015]

print("Titles added per year (2015 → ):")
print(trend.to_string())

# YoY growth
total_per_year = trend.sum(axis=1)
growth = total_per_year.pct_change() * 100
print("\nYoY growth rate:")
print(growth.dropna().map(lambda x: f'{x:.1f}%').to_string())
